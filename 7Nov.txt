def hitung_tarif(asal, tujuan, kelas, jarak):
  """Fungsi untuk menghitung tarif tiket kereta api.

  Args:
    asal: Stasiun keberangkatan.
    tujuan: Stasiun tujuan.
    kelas: Kelas kereta (ekonomi, bisnis, eksekutif).
    jarak: Jarak tempuh (dalam kilometer).

  Returns:
    Tarif tiket kereta api.
  """

  # Tarif dasar per kilometer untuk setiap kelas
  tarif_dasar = {
    "ekonomi": 1000,
    "bisnis": 1500,
    "eksekutif": 2000
  }

  # Hitung tarif dasar total
  tarif_total = tarif_dasar[kelas] * jarak

  # Tambahkan logika diskon di sini (opsional)
  # Contoh:
  # if usia < 12:
  #   tarif_total *= 0.5  # Diskon 50% untuk anak-anak

  return tarif_total

# Input data dari pengguna
asal = input("Stasiun Asal: ")
tujuan = input("Stasiun Tujuan: ")
kelas = input("Kelas (ekonomi/bisnis/eksekutif): ")
jarak = float(input("Jarak (km): "))

# Hitung dan tampilkan tarif
tarif = hitung_tarif(asal, tujuan, kelas, jarak)
print("Tarif tiket:", tarif)



def hitung_tarif_parkir(jenis_kendaraan, durasi):
  """Fungsi untuk menghitung tarif parkir.

  Args:
    jenis_kendaraan: Jenis kendaraan (mobil, motor, dll.).
    durasi: Durasi parkir (dalam jam).

  Returns:
    Tarif parkir.
  """

  # Tarif dasar per jam untuk setiap jenis kendaraan
  tarif_dasar = {
    "mobil": 5000,
    "motor": 3000
  }

  # Hitung tarif dasar total
  tarif_total = tarif_dasar[jenis_kendaraan] * durasi

  # Tambahkan logika diskon di sini (opsional)
  # Contoh:
  # if jam_masuk >= 18:
  #   tarif_total *= 0.8  # Diskon 20% setelah jam 6 sore

  return tarif_total

# Input data dari pengguna
jenis_kendaraan = input("Jenis Kendaraan (mobil/motor): ")
durasi = float(input("Durasi Parkir (jam): "))

# Hitung dan tampilkan tarif
tarif = hitung_tarif_parkir(jenis_kendaraan, durasi)
print("Tarif parkir:", tarif)

def hitung_harga_tiket(harga_dasar, jumlah_tiket, kode_diskon=None):
  """Fungsi untuk menghitung harga tiket konser

  Args:
    harga_dasar: Harga dasar satu tiket (int)
    jumlah_tiket: Jumlah tiket yang dibeli (int)
    kode_diskon: Kode diskon (str, opsional)

  Returns:
    Total harga tiket (int)
  """

  total_harga = harga_dasar * jumlah_tiket
  diskon = 0

  if kode_diskon:
    # Contoh perhitungan diskon berdasarkan kode
    if kode_diskon == 'DISKON10':
      diskon = total_harga * 0.1
    elif kode_diskon == 'DISKON20':
      diskon = total_harga * 0.2

  total_harga -= diskon
  return total_harga

# Input data dari pengguna
harga_dasar = int(input("Masukkan harga dasar tiket: "))
jumlah_tiket = int(input("Masukkan jumlah tiket yang dibeli: "))
kode_diskon = input("Masukkan kode diskon (jika ada): ")

# Hitung dan tampilkan total harga
total_harga = hitung_harga_tiket(harga_dasar, jumlah_tiket, kode_diskon)
print("Total harga tiket: Rp", total_harga)
